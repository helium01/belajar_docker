DOCKER DESKTOP ORACLE-XE BELAJAR 
eksekusi docker : docker exec -it [nama dockermu] *hilangi kurung
login bash : sqlplus sys/password as sysdba atau sqlplus / as sysdba
password : passwordmu

1. SHOW CON_NAME;
2. SHOW PDBS;
3. ALTER SESSION SET CONTAINER = XEPDB1;
4. 
CREATE TABLESPACE belajar_tables_space
DATAFILE 'belajar_tables_space.dbf' SIZE 50M
AUTOEXTEND ON;
5. 
CREATE USER usermu
IDENTIFIED BY passwordmu
DEFAULT TABLESPACE belajar_tables_space
TEMPORARY TABLESPACE temp
QUOTA UNLIMITED ON belajar_tables_space;
6. 
GRANT CREATE SESSION TO usermu;

-- Untuk buat tabel, view, dll
GRANT CREATE TABLE, CREATE VIEW, CREATE SEQUENCE, CREATE PROCEDURE, CREATE TRIGGER TO usermu;

-- Hak akses read/write semua tabel dalam PDB
GRANT SELECT ANY TABLE, INSERT ANY TABLE, UPDATE ANY TABLE, DELETE ANY TABLE TO usermu;

-- (Opsional) jika mau full admin di PDB
GRANT DBA TO usermu;
